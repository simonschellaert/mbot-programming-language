_StmtSeq_    := _StmtSeq_ *⏎* _Stmt_ | _Stmt_
_Stmt_       := _Assignment_ | _While_ | _If_ | _Command_

_Assignment_ := _Variable_ ⏪ _AExp_
_While_      := 🔁 _BExp_ *⇥* _StmtSeq_ *⇤*
_If_         := ❓ _BExp_ *⇥* _StmtSeq_ *⇤*

_AExp_       := _AExp_ + _ATerm_ | _AExp_ - _ATerm_ | _ATerm_
_ATerm_      := _ATerm_ * _AFactor_ | _ATerm_ / _AFactor_ | _AFactor_
_AFactor_    := (_AExp_) | Const | Variable

_BExp_       := _BExp_ barbar _BExp_ | _BTerm_
_BTerm_      := _BTerm_ && _BTerm_ | _BFactor_
_BFactor_    :=  👍  | 👎| ! _BFactor_ | (_BExp_) |  _BRel_
_BRel_       := _AExp_ < _AExp_ | _AExp_ == _AExp_ | _AExp_ > _AExp_

_Command_    := 💨 _Direction_ | 😴 _Duration_ | 🚨 _Flank_ _AExp_ _AExp_ _AExp_
_Direction_  := ⬇️ | ➡️ | ⬅️ | ⬆️
_Duration_   := 🕑
